  var chart = AmCharts.makeChart("chartdiv", {

    "type": "pie",

    "dataProvider": [{

      "domaine": "a",

      "valeurs": 501

    }, {

      "domaine": "b",

      "valeurs": 301

    }, {

      "domaine": "c",

      "valeurs": 201

    }, {

      "domaine": "d",

      "valeurs": 165

    }, {

      "domaine": "e",

      "valeurs": 139

    }, {

      "domaine": "f",

      "valeurs": 128

    }],

    "valueField": "valeurs",

    "titleField": "domaine"

  });


  function updateChart() {

    chart.dataProvider = [];

    for (var i = 0; i <= 9; i++) {

      if (Number(document.getElementById('value-' + i).value)) {

        chart.dataProvider.push({

          "domaine": document.getElementById('title-' + i).value,

          "valeurs": Number(document.getElementById('value-' + i).value)

        })

      }

    }

    chart.validateData();

  }

</script>
